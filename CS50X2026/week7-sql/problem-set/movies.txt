SQLS

1) SELECT title FROM movies
WHERE year = 2008;

2) SELECT birth FROM people
WHERE name = 'Emma Stone'

3) SELECT title FROM movies
WHERE year >= 2018
ORDER by title;

4) SELECT COUNT(*) as quantity_movies FROM ratings
WHERE rating = 10.0

5) SELECT title, year FROM movies
WHERE title LIKE 'Harry Potter and the%'
ORDER by year;

6) SELECT AVG(rating) FROM ratings
JOIN movies ON ratings.movie_id = movies.id
WHERE movies.year = 2012

7) SELECT rating, title FROM movies
JOIN ratings ON movies.id = ratings.movie_id
WHERE movies.year = 2010
ORDER BY rating DESC, title ASC

8) SELECT name FROM people
WHERE id IN (
    SELECT person_id FROM stars
    WHERE movie_id = (
           SELECT id FROM movies
           WHERE movies.title = 'Toy Story'
    )
);

9) SELECT name, id FROM people
WHERE id IN (
    SELECT person_id FROM stars
    WHERE movie_id IN (
        SELECT id FROM movies
        WHERE movies.year = 2004
    )
)
ORDER BY people.birth, people.name

10) SELECT name FROM people
WHERE id IN (
    SELECT person_id FROM directors
    WHERE movie_id IN (
        SELECT id FROM movies
        WHERE movie_id IN (
            SELECT movie_id FROM ratings
            WHERE rating >= 9.0
        )
    )
);

11) SELECT title FROM movies
WHERE id IN (
    SELECT movie_id FROM ratings
    WHERE movie_id IN (
        SELECT movie_id FROM stars
        WHERE person_id = (
            SELECT id from people
            WHERE name = 'Chadwick Boseman'
        )
    )
)
ORDER BY (
    SELECT rating
    FROM ratings
    WHERE ratings.movie_id = movies.id
) DESC
LIMIT 5

12) SELECT title FROM movies
WHERE id IN (
    SELECT movie_id FROM stars
    WHERE person_id IN (
        SELECT id FROM people
        WHERE name IN ('Jennifer Lawrence', 'Bradley Cooper')
    )
    GROUP BY movie_id
    HAVING COUNT(DISTINCT person_id) = 2
)

13) SELECT name FROM people
WHERE id IN (
    SELECT person_id FROM stars
    WHERE movie_id IN (
        SELECT movie_id FROM stars
        WHERE person_id = (
            SELECT id FROM people
            WHERE name = 'Kevin Bacon'
        )
    )
)
AND name != 'Kevin Bacon';
